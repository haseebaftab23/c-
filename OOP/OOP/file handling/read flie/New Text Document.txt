#include<iostream.h>
#include<conio.h>
#include<fstream>
#include<stdlib.h>
#include<string.h>
using namespace std;
class ReadCity
{
	string City;
	public:
	void rCity()
	{
		ifstream file("f:\\r\\city.txt");
		if(!file)
		{
			cout<<"File Is Not Present";
			exit(1);
		}
		while(!file.eof())
		{
			cout<<"City Name\n";
			file>>City;
			cout<<City<<"\n";
		}
	}
};
int main()
{
	ReadCity obj;
	obj.rCity();
	return 0;
	getch(); 
}
