#include<iostream.h>
#include<conio.h>
#include<string.h>
using namespace std;
class Publication
{
	private:
	string title;
	int id;
	public:
	void getData()
	{
		cout<<"\a\a\aEnter The Title :";
		cin>>title;
		cout<<"\a\a\aEnter The ID :";
		cin>>id;
	}
	void putData()
	{
		cout<<"Title :"<<title<<'\n';
		cout<<"ID :"<<id<<'\n';
	}
};
class Book:private Publication
{
	private:
	int pages;
	public:
	void getData()
	{
		Publication::getData();
		cout<<"\a\a\aEnter The Pages :";
		cin>>pages;
	}
	void putData()
	{
		Publication::putData();
		cout<<"Pages :"<<pages<<'\n';
	}
};
class Tape:private Book
{
	private:
	float ptime;
	public:
	void getData()
	{
		Book::getData();
		cout<<"\a\a\aEnter The Ptime :";
		cin>>ptime;
	}
	void putData()
	{
		Book::putData();
		cout<<"Ptime :"<<ptime<<'\n';
	}
};
main()
{
	Tape P;
	P.getData();
	cout<<"\n\n\n\n\n";
	P.putData();
	getch();
}
