#include<iostream.h>
#include<conio.h>
const int limit=10;
class safearray
{
	int arr[limit];
	public:
	int& access(int n)
	{
		if(n<0 || n>=limit)
		{
			cout<<"Index Is Out Of Bound";
	    	exit(1);			
		}
	    else
     	{
	    	return arr[n];
	    }
	}
	int access(int n)const 
	{
		if(n<0 || n>=limit)
		{
			cout<<"Index Is Out Of Bound";
			exit(1);
		}
		else
		{
			return arr[n];
		}
	}	
};
main()
{
	safearray s;
	for(int i=0; i<limit; i++)
     	s.access(i)=i*10;
	for(int j=0; j<limit; j++)
	{
		int temp;
		temp=s.access(j);
		cout<<"Element "<<j<<" Is "<<temp<<endl;
	}
getch();
}
