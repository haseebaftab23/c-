//Write a program that overloads relational operator to work with user-defined objects.

#include<iostream.h>
#include<conio.h>
class relational
{
	private:
	int feet;
	int inches;
	public:
	relational()
	{
		feet=0;
		inches=0;
	}
	relational(int f, int i)
	{
		feet=f;
		inches=i;
	}
	void display()
	{
		cout<<"\nFeet : "<<feet<<"\n";
		cout<<"Inches : "<<inches<<"\n";
	}
	void operator+=(relational);                  // relational than operator with "void operator+=(class name)"
};
void relational::operator+=(relational rel)      // similarly relational operator with "void operator-=(class name)"
{                                                   
	feet+=rel.feet;
	inches+=rel.inches;		
}
main()
{
	relational a1(1,2),b1(3,4);
	a1.display();
	b1.display();
	a1+=b1;
	a1.display();
	getch();
}
